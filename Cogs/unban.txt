from discord import Member
from discord.ext import commands
import json


class UnbanCog(commands.Cog, name="unban command"):
	def __init__(self, bot:commands.bot):
		self.bot = bot

	@commands.command(name = "unban",
					usage="",
					description = "unban a user.")
	@commands.event
	async def on_command_error(ctx, error):
		if isinstance(error, commands.MissingRequiredArgument):
			await ctx.send('Please pass in all requirements :rolling_eyes:.')
		if isinstance(error, commands.MissingPermissions):
			await ctx.send("You dont have all the permissions :no_entry_sign:")
	#The below code unbans player.
	@commands.cooldown(1, 2, commands.BucketType.member)
	@commands.has_permissions(Administrator = True)
	#The below code unbans player.
	async def unban(ctx, *, member):
		with open("Cogs/commands_config.json", "r") as config:
			data = json.load(config)
			quote_config = data["unban"]
			if quote_config[0] == "0":	
				DeleteMessage = False
			else:
				DeleteMessage = True
		banned_users = await ctx.guild.bans()
		member_name, member_discriminator = member.split("#")
		for ban_entry in banned_users:
			user = ban_entry.user
			message = await ctx.send(f"{member} was unbanned.")
		if DeleteMessage:
			await ctx.message.delete()

def setup(bot:commands.Bot):
	bot.add_cog(UnbanCog(bot))
